\acsetup{
	list/display = used,
	list/template = descriptionCustom,
	pages/display = first,
	make-links = true
}

\NewAcroTemplate[list]{descriptionCustom}{%
	\acroheading
	\acropreamble
	\begin{description}
		\setlength\labelwidth{1.8cm}
		\setlength\itemindent{0.75cm}
		\acronymsmap{%
			\item[\acrowrite{short}\acroifT{alt}{/\acrowrite{alt}}]
			\acrowrite{list}%
			\acroifanyT{foreign,extra}{ (}%
			\acroifT{foreign}{\acrowrite{foreign}\acroifT{extra}{, }}%
			\acroifT{extra}{\acrowrite{extra}}%
			\acroifanyT{foreign,extra}{)}%
			\acropagefill
			\acropages
			{\acrotranslate{page}\nobreakspace}
			{\acrotranslate{pages}\nobreakspace}%
		}
	\end{description}
}
